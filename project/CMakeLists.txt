cmake_minimum_required(VERSION 2.8)
project( Motion_Detector )
find_package( OpenCV REQUIRED )
include_directories( ${OpenCV_INCLUDE_DIRS} )
add_executable( sequential sequential.cpp )
target_link_libraries( sequential ${OpenCV_LIBS} )
add_executable( parallel parallel.cpp )
target_link_libraries( parallel ${OpenCV_LIBS} )
add_executable( parallel_class parallel_class.cpp )
target_link_libraries( parallel_class ${OpenCV_LIBS} )
add_executable( parallel_with_locks parallel_with_locks.cpp )
target_link_libraries( parallel_with_locks ${OpenCV_LIBS} )
add_executable( ff ff.cpp )
target_link_libraries( ff ${OpenCV_LIBS} )
add_executable( ff_alt ff_alt.cpp )
target_link_libraries( ff_alt ${OpenCV_LIBS} )
set(CMAKE_CXX_FLAGS "-O3 -std=c++20 -pthread")
